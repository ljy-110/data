正则表达式

```js
var INPUTRULES = {
  email: /^([a-zA-Z0-9]+[_|\_|\.]?)*[a-zA-Z0-9]+@([a-zA-Z0-9]+[_|\_|\.]?)*[a-zA-Z0-9]+\.[a-zA-Z]{2,3}$/,//邮箱
  mobile: /^(((13[0-9]{1})|(15[0-9]{1})|(17[0-9]{1})|(18[0-9]{1})|(14[0-9]{1}))+\d{8})$/,//手机
  id: /^[1-9][0-9]{5}(19[0-9]{2}|200[0-9]|2010)(0[1-9]|1[0-2])(0[1-9]|[12][0-9]|3[01])[0-9]{3}[0-9xX]$/,
  realname: /^[\u4e00-\u9fa5]{2,9}$/,
  enterprise: /^[\u4e00-\u9fa5]{3,51}/,
  zipcode: /^[0-9]{6}$/,//邮编
  fixedTel: /^((0\d{2,3})-)?(\d{7,8})(-(\d{3,}))?$/,//固定电话
  password: /^(?![0-9]+$)(?![a-zA-Z]+$)[0-9A-Za-z]{6,18}$/,
  qrcode: /^[a-zA-Z0-9]{4}$/,
  url: /http(s)?:\/\/([\w-]+\.)+[\w-]+(\/[\w- .\/?%&=]*)?/,
  domain: /([\w-]+\.)+[\w-]+(\/[\w- .\/?%&=]*)?/,
  organizationCode: /^[a-zA-Z0-9\-]+$/, // 组织机构
};
```
