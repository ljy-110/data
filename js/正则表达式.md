常用正则表达式

```js
email: /^([a-zA-Z0-9]+[_|\_|\.]?)*[a-zA-Z0-9]+@([a-zA-Z0-9]+[_|\_|\.]?)*[a-zA-Z0-9]+\.[a-zA-Z]{2,3}$/,//邮箱
mobile: /^(((13[0-9]{1})|(15[0-9]{1})|(17[0-9]{1})|(18[0-9]{1})|(14[0-9]{1}))+\d{8})$/,//手机
id: /^[1-9][0-9]{5}(19[0-9]{2}|200[0-9]|2010)(0[1-9]|1[0-2])(0[1-9]|[12][0-9]|3[01])[0-9]{3}[0-9xX]$/,//营业执照
realname: /^[\u4e00-\u9fa5]{2,9}$/,
enterprise: /^[\u4e00-\u9fa5]{3,51}/,
zipcode: /^[0-9]{6}$/,//邮编
fixedTel: /^((0\d{2,3})-)?(\d{7,8})(-(\d{3,}))?$/,//固定电话
password: /^(?![0-9]+$)(?![a-zA-Z]+$)[0-9A-Za-z]{6,18}$/,//密码
qrcode: /^[a-zA-Z0-9]{4}$/,//验证码
url: /http(s)?:\/\/([\w-]+\.)+[\w-]+(\/[\w- .\/?%&=]*)?/,//路由url
domain: /([\w-]+\.)+[\w-]+(\/[\w- .\/?%&=]*)?/,
organizationCode: /^[a-zA-Z0-9\-]+$/, // 组织机构
IDcard:/^(^[1-9]\d{7}((0\d)|(1[0-2]))(([0|1|2]\d)|3[0-1])\d{3}$)|(^[1-9]\d{5}[1-9]\d{3}((0\d)|(1[0-2]))(([0|1|2]\d)|3[0-1])((\d{4})|\d{3}[Xx])$)$/ //身份证验证
name：/[\u4e00-\u9fa5]/gm //匹配中文字符
string：/[^\x00-\xff]/igm //匹配双字节字符
number：/^\d+$/ //数字
number.length:/^\d{n}$/ //规定数字长度
number.min:/^\d{n,}$/  //最少输入n个数字
number.min.max:/^\d{m,n}$/ //规定长度的数字
English：/^[a-z]+$/i //只能输入英文
English.max:/^[A-Z]+$/ //大写英文字母组成
number.english:/^[a-z0-9]+$/i //英文和数字组成
num.eng.xia:/^\w+$/ //英文数字下划线组成
date:/^[1-2][0-9][0-9][0-9]-[0-1]{0,1}[0-9]-[0-3]{0,1}[0-9]$/  //日期 yyyy-mm-dd
```
